\noindent\textbf{Задача 5.} Доказать, что любое отношение $R$, симметричное и антисмметричное одновременно, является транзитвным.\\

\noindent\underline{Решение:}
\begin{flalign*}
&\text{Рассмотри пару} <x,y>\ \in R \Rightarrow \text{по симметричности }<y,x>\ \in R &\\
&\Rightarrow \text{по антисимметричности } x = y \Rightarrow\ <x,x>\ \in R &\\ 
&\Rightarrow R \text{ является транзитивным}&
\end{flalign*}
